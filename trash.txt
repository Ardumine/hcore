using System.Reflection;
using System.Runtime.Loader;
using HCore.Main.Fs;
using HCore.Main.Fs.Cmp;
using HCore.Modules.Base;
using Logyt;

namespace HCore.Main;

internal static class ProgramA
{
    private static readonly List<ModPack> ModPacks = [];
    private static HCoreFs _fs = null!;

    //https://stackoverflow.com/questions/40384619/how-to-load-assembly-from-stream-in-net-core
    private static Assembly LoadAssembly(Stream peStream, Stream pdbStream)
    {
        return AssemblyLoadContext.Default.LoadFromStream(peStream, pdbStream);
    }

    private static List<IModuleLoadDescriptor> GetModulesFromModPack(ModPack modPack)
    {
        var descriptors = new List<IModuleLoadDescriptor>();

        var moduleDllStream = _fs.GetFileStreamRead(Path.Combine(modPack.Path, $"{modPack.Name}.dll"));
        var modulePdbStream = _fs.GetFileStreamRead(Path.Combine(modPack.Path, $"{modPack.Name}.pdb"));

        var assembly = LoadAssembly(moduleDllStream, modulePdbStream);
        moduleDllStream.Dispose();
        modulePdbStream.Dispose();

        var types = assembly.GetTypes();

        foreach (var type in types)
            if (typeof(IModuleLoadDescriptor).IsAssignableFrom(type))
                if (Activator.CreateInstance(type) is IModuleLoadDescriptor result)
                    descriptors.Add(result);

        return descriptors;
    }

    private static void ImportModuleDescriptorsIntoFs(ModPack modPack)
    {
        // Import each descriptor in the /mods/ folder
        const string modDescriptorsPath = "/mods/";
        foreach (var descriptor in modPack.ModuleDecriptors)
        {
            _fs.CreateCmp(
                path: Path.Join(modDescriptorsPath, $"{descriptor.Name}"),
                cmp: new CmpModulePointer(
                    moduleName: descriptor.Name,
                    packName: modPack.Name
                )
            );
        }
    }

    private static void LoadModPacks()
    {
        foreach (var dir in _fs.ListDirs("/packs"))
        {
            // Discover available modules
            var modPack = new ModPack
            {
                Name = new DirectoryInfo(dir).Name,
                Path = dir,
                ModuleDecriptors = new List<IModuleLoadDescriptor>()
            };

            modPack.ModuleDecriptors.AddRange(GetModulesFromModPack(modPack));
            ImportModuleDescriptorsIntoFs(modPack);

            ModPacks.Add(modPack);
        }
    }

    private static IModuleLoadDescriptor? GetModuleDescriptorFromFile(string path)
    {
        var cmp = _fs.ReadEntry<CmpModulePointer>(path);
        if (cmp == null)
            throw new Exception("The file is not a CMP pointer");

        var modPack = ModPacks.Find(x => x.Name == cmp.PackName);
        return modPack?.ModuleDecriptors.SingleOrDefault(x => x.Name == cmp.ModuleName);
    }

    private static BaseImplement? CreateModuleInstance(Type implementType)
    {
        return Activator.CreateInstance(implementType) as BaseImplement;
    }

    private static IInitModule GetAndCreateInitModule(string modPath = "/hcore/init")
    {
        // Get the module descriptor
        var modDesc = GetModuleDescriptorFromFile(modPath);
        if (modDesc == null) throw new Exception($"Could not find module descriptor for {modPath}");

        // Initiate an instance of the module
        var instance = CreateModuleInstance(modDesc.ImplementType);

        if (instance == null)
            throw new Exception($"Could not create instance for {modDesc.ImplementType}");

        if (instance is not IInitModule initModule)
            throw new Exception($"Module {modDesc.ImplementType} is not an init module");

        return initModule;
    }


    private static void MainA()
    {
        // Create base logger
        var logyt = new ConsoleLogyt("HCore");

        // Mount
        logyt.I("Loading fs...");
        _fs = new HCoreFs("/home/ardumine/hort/hcore/FS");

        // Load mod packs
        logyt.I("Loading modpacks...");
        LoadModPacks();

        // Get the init module
        var initModule = GetAndCreateInitModule("/mods/HCore.Modules.TestDemo.Module2");

        // Run init
        logyt.I("Running init...");
        initModule.Run();

        // After the init module finishes, the HCore will stop
        logyt.I("HCore done!");
    }
}

public static class ProgramVfs
{
    public static void MainVfs()
    {
        var vfs = new FileSystem();

        vfs.Mount("/", new MemoryFileSystem());
        vfs.Mount("/dev", new DeviceFileSystem());
        vfs.Mount("/realfs", new HostFileSystem("/home/ardumine/"));


        bool exec = true;
        string currentPath = "/";
        while (exec)
        {
            // Escrever linha

            Console.Write($"{currentPath}$ ");

            var line = Console.ReadLine();
            var splitLine = line.Split(" ");
            var command = splitLine[0];

            if (command == "mkdir")
            {
                var folderName = splitLine[1];
                vfs.MkDir(folderName);
            }
            else if (command == "cat")
            {
                var fileName = splitLine[1];
                var file = vfs.GetFile(fileName);
                Console.WriteLine(file.ReadString());
            }
        }
    }
}
